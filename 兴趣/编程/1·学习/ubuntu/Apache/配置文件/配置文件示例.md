```
<VirtualHost *:80>
    ServerAdmin webmaster@localhost
    ServerName 10.130.35.137
    
    # UIPreview 的主配置
    DocumentRoot /var/www/uipreview
    <Directory /var/www/uipreview>
        Options Indexes FollowSymLinks MultiViews
        AllowOverride All
        Require all granted
    </Directory>

    # image-manager 前端配置
    Alias /img /var/www/image-manager/image-manager-frontend
    <Directory /var/www/image-manager/image-manager-frontend>
        Options Indexes FollowSymLinks
        AllowOverride All
        Require all granted

        # Vue 路由配置
        RewriteEngine On
        RewriteBase /img/
        RewriteRule ^index\.html$ - [L]
        RewriteCond %{REQUEST_FILENAME} !-f
        RewriteCond %{REQUEST_FILENAME} !-d
        RewriteRule . /img/index.html [L]
    </Directory>

    # image-manager API 代理配置
    <Location /img/api>
        ProxyPass http://localhost:3002/api
        ProxyPassReverse http://localhost:3002/api
    </Location>

    <Location /img/images>
        ProxyPass http://localhost:3002/images
        ProxyPassReverse http://localhost:3002/images
    </Location>

    # temppix 配置
    Alias /temppix /var/www/temppix/public
    <Directory /var/www/temppix/public>
        Options Indexes FollowSymLinks
        AllowOverride All
        Require all granted

        # 单页应用重写规则
        RewriteEngine On
        RewriteBase /temppix/
        RewriteRule ^index\.html$ - [L]
        RewriteCond %{REQUEST_FILENAME} !-f
        RewriteCond %{REQUEST_FILENAME} !-d
        RewriteRule . /temppix/index.html [L]
    </Directory>

    # temppix uploads 静态文件配置
    Alias /temppix/uploads /var/www/temppix/uploads
    <Directory /var/www/temppix/uploads>
        Options Indexes FollowSymLinks
        AllowOverride None
        Require all granted
    </Directory>

    # temppix API 代理配置
    <Location /temppix/api>
        ProxyPass http://localhost:3003/api
        ProxyPassReverse http://localhost:3003/api
    </Location>

    <Location /temppix/images>
        ProxyPass http://localhost:3003/images
        ProxyPassReverse http://localhost:3003/images
    </Location>

    # 日志配置
    ErrorLog ${APACHE_LOG_DIR}/error.log
    CustomLog ${APACHE_LOG_DIR}/access.log combined
</VirtualHost> 
```
